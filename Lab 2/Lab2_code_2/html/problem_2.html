%--------------------------------------------------------------------------
% Define the piecewise function
function x = piecewise(t)
    if t >= -5 && t < 5 % if statement; [-5,5) then:
        x = -2.*abs(t) + 10; % y = -2|t| + 10
    elseif t >= 5 && t < 10 %if statement [5,10) then:
        x = 10;
    else
        x = 0;
    end
end
